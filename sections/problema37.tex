	\item $AxC$ = $\{\{a,3\},\{a,4\},\{b,3\},\{b,4\}\}$

	\item $A x B x C$ = $\{\{\{a,2\},3\},\{\{a,3\},3\},\{\{b,2\},3\},\{\{b,3\},3\}\}$
\end{itemize}
